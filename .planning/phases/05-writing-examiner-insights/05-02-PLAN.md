<objective>
Create the IELTS→DSE conversion mapping table and wire it into the correction pipeline — replacing the current percentage-based DSE level calculation with IELTS-band-based conversion.

**Purpose:** The conversion table provides the calibrated bridge between IELTS bands (which LLMs score accurately) and DSE levels (which students need). By making the table configurable in localStorage (same pattern as existing DSE boundaries), power users can fine-tune the calibration. The combineCorrections function now derives DSE level from IELTS bands, not from raw Content/Org/Lang percentage, eliminating inflated grades.

**Output:**
- dseGrading.js: conversion functions + default mapping table + localStorage persistence
- useDSEPapers.js: combineCorrections wired to use IELTS→DSE conversion, validateCorrectionOutput with IELTS checks
</objective>

```javascript
// IELTS Writing band → DSE level conversion mapping
// Calibrated from HKEAA level descriptors and examiner report data.
// Multiple IELTS bands can map to the same DSE level because IELTS
// has finer granularity (0.5 increments) than DSE (7 levels).
const IELTS_TO_DSE_KEY = 'crescendo-ielts-dse-map';

const IELTS_TO_DSE_DEFAULTS = [
  { ieltsBand: 9.0, dseLevel: '5**', minDsePct: 95 },
  { ieltsBand: 8.5, dseLevel: '5**', minDsePct: 90 },
  { ieltsBand: 8.0, dseLevel: '5*',  minDsePct: 84 },
  { ieltsBand: 7.5, dseLevel: '5',   minDsePct: 76 },
  { ieltsBand: 7.0, dseLevel: '4',   minDsePct: 67 },
  { ieltsBand: 6.5, dseLevel: '4',   minDsePct: 62 },
  { ieltsBand: 6.0, dseLevel: '3',   minDsePct: 55 },
  { ieltsBand: 5.5, dseLevel: '2',   minDsePct: 45 },
  { ieltsBand: 5.0, dseLevel: '2',   minDsePct: 38 },
  { ieltsBand: 4.5, dseLevel: '1',   minDsePct: 25 },
  { ieltsBand: 4.0, dseLevel: '1',   minDsePct: 0 },
];

export function getIeltsToDseMap() {
  try {
    const raw = localStorage.getItem(IELTS_TO_DSE_KEY);
    if (raw) {
      const parsed = JSON.parse(raw);
      if (Array.isArray(parsed) && parsed.length > 0 && parsed[0].ieltsBand !== undefined) {
        return parsed;
      }
    }
  } catch {}
  return IELTS_TO_DSE_DEFAULTS.map(e => ({ ...e }));
}

export function storeIeltsToDseMap(map) {
  localStorage.setItem(IELTS_TO_DSE_KEY, JSON.stringify(map));
}

export function resetIeltsToDseMap() {
  storeIeltsToDseMap(IELTS_TO_DSE_DEFAULTS);
  return IELTS_TO_DSE_DEFAULTS.map(e => ({ ...e }));
}

export function ieltsBandsToDseLevel(overallBand, criteriaBands) {
  const map = getIeltsToDseMap();
  // Sort descending by ieltsBand
  const sorted = [...map].sort((a, b) => b.ieltsBand - a.ieltsBand);
  for (const entry of sorted) {
    if (overallBand >= entry.ieltsBand) {
      return entry.dseLevel;
    }
  }
  return '1';
}

export function ieltsBandToPct(band) {
  const map = getIeltsToDseMap();
  const entry = map.find(e => e.ieltsBand === band);
  if (entry) return entry.minDsePct;
  // Interpolate between nearest bands
  const sorted = [...map].sort((a, b) => b.ieltsBand - a.ieltsBand);
  for (let i = 0; i < sorted.length - 1; i++) {
    if (band > sorted[i + 1].ieltsBand && band < sorted[i].ieltsBand) {
      const frac = (band - sorted[i + 1].ieltsBand) / (sorted[i].ieltsBand - sorted[i + 1].ieltsBand);
      return Math.round(sorted[i + 1].minDsePct + frac * (sorted[i].minDsePct - sorted[i + 1].minDsePct));
    }
  }
  return 0;
}
```

```javascript
if (typeof ieltsOverallBand === 'number') {
  const errorCount = result.errors?.length || 0;
  if (ieltsOverallBand >= 9 && errorCount > 0) {
```
